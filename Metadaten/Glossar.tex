  description={EXPRESS ist ein Standard für eine generische Datenmodellierungssprache für Produktdaten.
   EXPRESS ist im ISO-Standard für den Austausch von Produktmodelldaten STEP (ISO 10303) formalisiert und als ISO 10303-11 standardisiert.}
  }
\newglossaryentry{API}{
  name={API},
  description={Eine Schnittstelle, die es zwei oder mehr Programmen ermöglicht, miteinander zu kommunizieren (Application Programming Interface).}
}

\newglossaryentry{Cors}{
  name={CORS},
  description={Ein Sicherheitsmechanismus, der den Zugriff auf Ressourcen von einem anderen Ursprung (Domain, Protokoll, Port) regelt (Cross-Origin Resource Sharing).}
}

\newglossaryentry{Framework}{
  name={Framework},
  description={Ein vorgefertigtes Gerüst, das die Entwicklung von Software vereinfacht.}
}

\newglossaryentry{HLS}{
  name={HLS},
  description={Ein von Apple entwickeltes Protokoll zur Übertragung von Video- und Audio-Streams (HTTP Live Streaming).}
}

\newglossaryentry{JSON}{
  name={JSON},
  description={Ein leichtgewichtiges Datenformat, das zum Austausch von Daten zwischen Server und Client verwendet wird (JavaScript Object Notation).}
}

\newglossaryentry{MJPEG}{
  name={MJPEG},
  description={Ein Videoformat, bei dem jeder Frame als separates JPEG-Bild codiert ist (Motion JPEG).}
}

\newglossaryentry{OpenAPI}{
  name={OpenAPI},
  description={Ein Standard für die Beschreibung von REST-APIs, oft im JSON- oder YAML-Format.}
}

\newglossaryentry{PWA}{
  name={Progressive Web App},
  description={Eine Webanwendung, die wie eine native App aussieht und funktioniert, aber mit Webtechnologien erstellt wurde. (Progressive Web App).}
}

\newglossaryentry{PTZ}{
  name={PTZ},
  description={Eine Funktion bei Kameras, die es ermöglicht, die Kamera horizontal (pan), vertikal (tilt) zu schwenken und zu zoomen (Pan-Tilt-Zoom).}
}



\newglossaryentry{Reverse Proxy}{
  name={Reverse Proxy},
  description={Ein Server, der Anfragen von Clients empfängt und an einen oder mehrere andere Server im internen Netzwerk weiterleitet.}
}

\newglossaryentry{SSE}{
  name={Server Sent Events},
  description={Ein Mechanismus, der es einem Server ermöglicht, Updates an einen Browser zu senden.}
}

\newglossaryentry{Uuid}{
  name={UUID},
  description={Eine 128-Bit-Zahl, die zur eindeutigen Identifizierung von Informationen dient (Universally Unique Identifier).}
}

\newglossaryentry{WebApp}{
  name={Webanwendung},
  description={Eine Anwendung, die über einen Webbrowser zugänglich ist.}
}
\newglossaryentry{Grafik-Engine}{
    name={Grafik-Engine},
    description={Eine Grafik-Engine (auch Render-Engine genannt) ist eine \\ Software-Komponente, die für die Darstellung von 2D- oder 3D-Grafiken verantwortlich ist. Ihre Hauptaufgabe ist es, aus geometrischen Daten, Texturen und Lichtinformationen ein sichtbares Bild zu generieren. 
    }
}
\newglossaryentry{Bibliothek}{
    name={Bibliothek},
    description={ Eine Bibliothek ist eine Sammlung von Funktionen oder Klassen, die eine Programmiersprache um weitere Befehle erweitert. 
    }
}

\newglossaryentry{Rest-API}{
    name={Rest-API},
    description={
    Ein Architekturstil für verteilte Systeme, der häufig für die Entwicklung von Web-APIs genutzt wird (Representational State Transfer).
    }
}
\newglossaryentry{Plugin}{
    name={Plugin},
    description={
    Ein Plugin ist eine Softwareerweiterung, die eine Anwendung um zusätzliche Funktionalität erweitert, ohne den Quellcode der Hauptanwendung zu ändern.
    }
}
\newglossaryentry{Bubblewrap}{
    name={Bubblewrap},
    description={
        Ein von Google entwickeltes Befehlszeilen-Tool, das Progressive Web Apps (PWAs) in native Android-Apps verpackt. Es automatisiert die Erstellung einer Trusted Web Activity (TWA), um eine PWA für die Veröffentlichung im Google Play Store vorzubereiten. Es liest die Metadaten aus dem Web-App-Manifest der PWA und generiert ein Android-Projekt, das die PWA als eigenständige Anwendung starten kann.
    }
}

\newglossaryentry{GooglePlayStore}{
    name={Google Play Store},
    description={
        Der offizielle App-Store von Google für das Android-Betriebssystem.
    }
}
\newglossaryentry{AppleAppStore}{
    name={Apple App Store},
    description={
        Der offizielle App-Store von Apple für die Betriebssysteme iOS, iPadOS, macOS, watchOS und tvOS. Entwickler können hier ihre Anwendungen veröffentlichen, die vor der Freigabe einem strengen Überprüfungsprozess von Apple unterzogen werden.
    }
}
\newglossaryentry{APK}{
    name={APK},
    description={
        Die Abkürzung für "Android Package Kit". Es handelt sich um ein Dateiformat, das von Android zur Verteilung und Installation mobiler Anwendungen verwendet wird. Eine APK-Datei ist im Wesentlichen ein Archiv, das alle Elemente enthält, die eine Anwendung für die korrekte Installation auf einem Android-Gerät benötigt, darunter der kompilierte Code, Ressourcen, Assets und die Manifest-Datei.
    }
}

\newglossaryentry{Webstorm}{
 name={WebStorm},
 description={Eine von JetBrains entwickelte integrierte Entwicklungsumgebung (IDE), die sich auf die Erstellung von JavaScript-Anwendungen spezialisiert hat.}
}
